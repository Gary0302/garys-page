Lang switch script (required): ```javascript const sections = document.querySelectorAll('.section,.doc-header'); const navLinks = document.querySelectorAll('.sidenav a[href^="#"]'); const langBtns = document.querySelectorAll('[data-set-lang]'); function setLang(lang) { document.body.dataset.lang = lang; document.documentElement.lang = lang === 'zh' ? 'zh-Hant' : 'en'; langBtns.forEach(b => b.classList.toggle('active', b.dataset.setLang === lang)); localStorage.setItem('garys-format-lang', lang); if (window.MathJax?.typesetPromise) MathJax.typesetPromise(); } langBtns.forEach(b => b.addEventListener('click', () => setLang(b.dataset.setLang))); setLang(localStorage.getItem('garys-format-lang') || 'en'); const obs = new IntersectionObserver(entries => { entries.forEach(e => { if (e.isIntersecting) { const id = e.target.id; navLinks.forEach(l => l.classList.toggle('active', l.getAttribute('href') === `#${id}`)); } }); }, { threshold: 0.2, rootMargin: '-60px 0px -40% 0px' }); sections.forEach(s => obs.observe(s)); ``` ## Typography rules ### Equations vs explanatory text (critical) | Content | Style | |---------|-------| | `$$...$$` display math | MathJax CHTML — do NOT override `mjx-container` | | `.formula-note` below equations | Body sans `.9375rem`, `color: --ink2`, no italic | | `.def-type` (ℤ₊, ℝ₊) | Mono `.875rem`, readable — not tiny caption size | ### Tables & matrices (readable size) | Element | Size | |---------|------| | `.tbl` / `.matrix` cells | `.875rem` (14px) | | `.tbl th` / `.matrix th` | `.75rem` (12px) | | `.matrix-label` | `.8125rem` with `line-height: 1.5` | | `.caption` | `.8125rem`, `line-height: 1.65`, no uppercase | ### Value colors (keep these — user likes them) -
