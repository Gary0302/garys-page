SNN N! Ordering Space
Experiment Design

三個循序漸進的實驗，從最基本的可行性出發，逐步建立對 ordering space 幾何結構的信心。每個實驗都有明確的輸入、流程、診斷工具與觀察重點。

核心設計原則

所有實驗共用同一個 RPC 映射（固定、無參數）。如果模型學得好，功勞只能來自 SNN 的 ordering；如果學不好，問題也只能在 SNN 或任務設計。變因控制是這套框架的靈魂。

驗證目標成功條件規模
E1Ordering 可分類模型確實用 ordering 做決策C=M=4
E2RPC 距離結構保留相似 class → 相近 RPC 座標C=M=4
E3N! 空間 scalingM 增大時容量有效增加M=4→20

Toy Dataset 設計

輸入為 one-hot vector X ∈ {0,1}^C。每個 class 對應一個獨熱維度，沒有雜訊，任務本身零歧義。這讓我們排除資料複雜度，確認 ordering 的學習是否發生。

資料結構

ClassOne-hot X（C=4）目標 ordering（示意）
class 0 [1, 0, 0, 0] 1234 → SNN 自行學習
class 1 [0, 1, 0, 0] 4321 → SNN 自行學習
class 2 [0, 0, 1, 0] 2413 → SNN 自行學習
class 3 [0, 0, 0, 1] 3142 → SNN 自行學習
重要：我們不指定目標 ordering

訓練只提供 class label，SNN 必須自己發現哪些 ordering pattern 能被 RPC 讀出為正確的 class。目標 ordering 欄位只是「如果學成功後可能長這樣」的示意，不是監督信號。

pythontoy_dataset.py
import torch
from torch.utils.data import Dataset, DataLoader

class OneHotOrderingDataset(Dataset):
    """
    最簡單的驗證資料集。
    輸入：one-hot vector（C 維）
    標籤：class index（0 到 C-1）
    完全確定性，沒有雜訊。
    """
    def __init__(self, C: int, n_per_class: int = 200):
        self.C = C
        self.n = n_per_class
        # 重複每個 one-hot 向量 n 次
        xs, ys = [], []
        for c in range(C):
            one_hot = torch.zeros(C)
            one_hot[c] = 1.0
            xs.append(one_hot.unsqueeze(0).expand(n_per_class, -1))
            ys.append(torch.full((n_per_class,), c, dtype=torch.long))
        self.X = torch.cat(xs)  # (C*n, C)
        self.y = torch.cat(ys)  # (C*n,)

    def __len__(self): return len(self.y)
    def __getitem__(self, i): return self.X[i], self.y[i]


def make_loaders(C=4, n_per_class=200, batch_size=32, train_ratio=0.8):
    ds = OneHotOrderingDataset(C, n_per_class)
    n_train = int(len(ds) * train_ratio)
    n_val   = len(ds) - n_train
    train_ds, val_ds = torch.utils.data.random_split(ds, [n_train, n_val])
    train_dl = DataLoader(train_ds, batch_size=batch_size, shuffle=True)
    val_dl   = DataLoader(val_ds,   batch_size=batch_size, shuffle=False)
    return train_dl, val_dl

觀察重點

不預設數字門檻。我們要觀察的是：

  • Loss 是否下降：loss curve 有無明顯的學習信號
  • Ordering 是否分化：不同 class 的 first_spike_time 分布是否開始不同
  • RPC 座標是否聚類：同一 class 的樣本在 RPC 空間中是否靠近（用 t-SNE 視覺化）
  • Firing rate 是否是替代路徑：檢查 spike count 是否在各 class 間相同（控制 spurious feature）

Spurious Feature 控制

最重要的診斷

計算每個 class 的平均 spike count（firing rate）。如果各 class 的 spike count 有明顯差異，模型可能透過 firing rate 而非 ordering 在分類。只有在 spike count 相近的情況下，ordering 的貢獻才能被確認。

pythonexp1_run.py
import torch, torch.nn as nn, torch.nn.functional as F
import snntorch as snn
from collections import defaultdict

def run_exp1(C=4, M=4, T=16, epochs=100, lr=1e-3):
    train_dl, val_dl = make_loaders(C=C)
    model    = SNNOrderingFramework(N=C, M=M, C=C, T=T)
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)

    log = defaultdict(list)

    for ep in range(epochs):
        model.train()
        ep_loss = 0

        for x, y in train_dl:
            optimizer.zero_grad()
            logits, S = model(x)               # S: (batch, M, T)
            loss = F.cross_entropy(logits, y)
            loss.backward()
            optimizer.step()
            ep_loss += loss.item()

        # ── 驗證 ──
        model.eval()
        correct, total = 0, 0
        spike_counts = defaultdict(list)  # 收集各 class 的 firing rate
        rpc_coords   = defaultdict(list)  # 收集各 class 的 RPC 座標

        with torch.no_grad():
            for x, y in val_dl:
                logits, S = model(x)
                pred = logits.argmax(dim=1)
                correct += (pred == y).sum().item()
                total   += len(y)

                # Spurious feature 診斷：收集 spike count per class
                sc = S.sum(dim=(1, 2))  # (batch,)：每個樣本的總 spike 數
                rc = rpc_embedding(S, T) # (batch, M-1)
                for i, c in enumerate(y.tolist()):
                    spike_counts[c].append(sc[i].item())
                    rpc_coords[c].append(rc[i].tolist())

        acc = correct / total
        log['loss'].append(ep_loss / len(train_dl))
        log['acc'].append(acc)
        log['spike_counts'].append(spike_counts)
        log['rpc_coords'].append(rpc_coords)

        if ep % 10 == 0:
            sc_means = {c: sum(v)/len(v) for c, v in spike_counts.items()}
            print(f"ep={ep:3d}  loss={ep_loss/len(train_dl):.4f}  acc={acc:.3f}  spike_per_class={sc_means}")

    return model, log

設計「有結構的相似性」

為了驗證距離結構，我們需要在 class 之間定義語意距離。最直接的方式：讓某些 class pair 共享更多輸入特徵，然後確認它們的 RPC 座標是否也更近。

pythonexp2_analysis.py
from sklearn.manifold import TSNE
import numpy as np

def analyze_rpc_geometry(model, val_dl, T):
    """
    收集所有樣本的 RPC 座標，計算：
    1. Within-class 平均距離（應小）
    2. Between-class 平均距離（應大）
    3. 輸出 t-SNE 投影座標供視覺化
    """
    model.eval()
    all_coords, all_labels = [], []

    with torch.no_grad():
        for x, y in val_dl:
            _, S = model(x)
            coords = rpc_embedding(S, T)     # (batch, M-1)
            all_coords.append(coords.numpy())
            all_labels.append(y.numpy())

    coords = np.concatenate(all_coords)   # (N, M-1)
    labels = np.concatenate(all_labels)   # (N,)

    # Within-class vs Between-class 距離
    C = len(np.unique(labels))
    class_centers = np.array([
        coords[labels == c].mean(axis=0) for c in range(C)
    ])  # (C, M-1)

    within_dist = np.mean([
        np.linalg.norm(coords[labels == c] - class_centers[c], axis=1).mean()
        for c in range(C)
    ])

    between_pairs = [
        np.linalg.norm(class_centers[i] - class_centers[j])
        for i in range(C) for j in range(i+1, C)
    ]
    between_dist = np.mean(between_pairs)

    print(f"Within-class dist  : {within_dist:.4f}  ← 應盡量小")
    print(f"Between-class dist : {between_dist:.4f}  ← 應盡量大")
    print(f"Separation ratio   : {between_dist/within_dist:.2f}x  ← 越大越好")

    # t-SNE（M-1=3 維時可直接用，更高維需要降維）
    if coords.shape[1] > 2:
        tsne_coords = TSNE(n_components=2, perplexity=20, random_state=42).fit_transform(coords)
    else:
        tsne_coords = coords

    return {
        'coords':       coords,
        'tsne_coords':  tsne_coords,
        'labels':       labels,
        'centers':      class_centers,
        'within_dist':  within_dist,
        'between_dist': between_dist,
    }

成功判讀

Separation ratio（between / within）是 E2 的核心指標。沒有固定門檻，但：

  • ratio > 1：基本的群集結構存在
  • ratio > 3：RPC 空間有清晰的幾何分離
  • t-SNE 可見明確群集：視覺確認，最直觀的成功標誌

實驗設計

對每個 M 值，訓練獨立模型，記錄收斂後的 accuracy 與 separation ratio。我們期望看到 M 增大時：

  • 相同 C 下，accuracy 維持或提升（更大的 ordering 空間 → 更鬆弛的分類壓力）
  • C=M 設定下，隨著 M 增大模型仍能學習（驗證 N! 空間確實被有效利用）
pythonexp3_scaling.py
import math

def run_exp3(M_list=None, epochs=100):
    if M_list is None:
        M_list = [4, 6, 8, 10, 12, 16, 20]

    results = []
    for M in M_list:
        C = M  # C = M：每增加一個 neuron，也增加一個 class
        T = max(16, M * 2)  # T 要夠大，讓所有 neuron 都有時間發放

        print(f"\n── M={M}, C={C}, T={T}, M!={math.factorial(M):,} ──")
        train_dl, val_dl = make_loaders(C=C, n_per_class=200)
        model = SNNOrderingFramework(N=C, M=M, C=C, T=T)
        model, log = run_exp1(C=C, M=M, T=T, epochs=epochs)

        geo = analyze_rpc_geometry(model, val_dl, T)
        results.append({
            'M':            M,
            'factorial':    math.factorial(M),
            'final_acc':    log['acc'][-1],
            'sep_ratio':    geo['between_dist'] / geo['within_dist'],
        })
        print(f"  → acc={results[-1]['final_acc']:.3f}  sep={results[-1]['sep_ratio']:.2f}x")

    return results

預期觀察與詮釋

現象詮釋
acc 隨 M 增大穩定維持N! 空間確實在承載分類資訊，scaling 有效
acc 在大 M 時下降SNN 訓練困難（梯度消失、spike 稀疏），不是 N! 空間的問題
sep_ratio 隨 M 增大而增大更大的空間讓不同 class 的 ordering 更容易分開

診斷工具集

這些工具在任何實驗階段都可以呼叫，用於理解模型的內部狀態。

pythondiagnostics.py
def diagnose_ordering(model, x_batch, T):
    """
    給定一個 batch，輸出：
    - 每個樣本的 first_spike_time（M 維向量）
    - 對應的 ordering（排列）
    - RPC 座標
    - spike count（用於 spurious feature 檢查）
    """
    model.eval()
    with torch.no_grad():
        _, S = model(x_batch)   # S: (batch, M, T)

    has_spike  = (S > 0)
    first_time = torch.where(
        has_spike.any(dim=-1),
        has_spike.float().argmax(dim=-1),
        torch.full(S.shape[:2], float(T))
    )  # (batch, M)

    ordering    = first_time.argsort(dim=-1)  # (batch, M)
    rpc_coord   = rpc_embedding(S, T)         # (batch, M-1)
    spike_count = S.sum(dim=(1, 2))           # (batch,)

    return {
        'first_time':  first_time,
        'ordering':    ordering,
        'rpc_coord':   rpc_coord,
        'spike_count': spike_count,
        'spike_matrix': S,
    }


def check_spurious(log, epoch=-1):
    """
    從 E1 的 log 中取出最後一個 epoch 的 spike_counts，
    計算各 class 的平均與標準差，判斷是否有 firing rate 洩漏。
    """
    sc = log['spike_counts'][epoch]
    for c, vals in sorted(sc.items()):
        mean = sum(vals) / len(vals)
        std  = ((sum((v-mean)**2 for v in vals) / len(vals))**0.5)
        print(f"class {c}: spike_count = {mean:.2f} ± {std:.2f}")


def visualize_ordering_diversity(model, val_dl, T, C):
    """
    對每個 class，收集所有樣本的 ordering，
    計算 ordering 多樣性（同一 class 內部的排列是否一致）。
    一致 → SNN 學到了穩定的 ordering pattern。
    """
    model.eval()
    class_orderings = defaultdict(list)
    with torch.no_grad():
        for x, y in val_dl:
            diag = diagnose_ordering(model, x, T)
            for i, c in enumerate(y.tolist()):
                class_orderings[c].append(diag['ordering'][i].tolist())

    for c in range(C):
        orderings = class_orderings[c]
        # 最常見的排列 + 出現頻率
        from collections import Counter
        top = Counter(map(tuple, orderings)).most_common(3)
        print(f"class {c}: top orderings = {top}")

"""
SNN N! Ordering Space — Full Experiment Suite
執行：python snn_ordering_full.py
需要：pip install torch snntorch scikit-learn
"""
import math, torch, torch.nn as nn, torch.nn.functional as F
import snntorch as snn
import numpy as np
from collections import defaultdict, Counter
from torch.utils.data import Dataset, DataLoader
from sklearn.manifold import TSNE

# ════════════════════════════════════════════
#  固定映射（無參數）
# ════════════════════════════════════════════
def rpc_embedding(S, T):
    batch, M, _ = S.shape
    has   = (S > 0)
    ft    = torch.where(has.any(dim=-1),
                has.float().argmax(dim=-1),
                torch.full((batch, M), float(T)))
    order = ft.argsort(dim=-1)
    st    = torch.gather(ft, 1, order)
    coords, rem = [], torch.full((batch,), float(T))
    for i in range(M - 1):
        c = (st[:, i] / (rem + 1e-8)).clamp(0, 1)
        coords.append(c)
        rem = (rem - st[:, i]).clamp(min=0)
    return torch.stack(coords, dim=-1)

# ════════════════════════════════════════════
#  Dataset
# ════════════════════════════════════════════
class OneHotDS(Dataset):
    def __init__(self, C, n=200):
        xs, ys = [], []
        for c in range(C):
            oh = torch.zeros(C); oh[c] = 1.0
            xs.append(oh.unsqueeze(0).expand(n, -1))
            ys.append(torch.full((n,), c, dtype=torch.long))
        self.X, self.y = torch.cat(xs), torch.cat(ys)
    def __len__(self): return len(self.y)
    def __getitem__(self, i): return self.X[i], self.y[i]

def make_loaders(C, n=200, bs=32):
    ds = OneHotDS(C, n)
    n_tr = int(len(ds) * 0.8)
    tr, va = torch.utils.data.random_split(ds, [n_tr, len(ds)-n_tr])
    return DataLoader(tr, bs, shuffle=True), DataLoader(va, bs)

# ════════════════════════════════════════════
#  Model
# ════════════════════════════════════════════
class SNNOrderingFramework(nn.Module):
    def __init__(self, N, M, C, T, b1=0.9, b2=0.95):
        super().__init__()
        self.T, self.M = T, M
        self.fc   = nn.Linear(N, M)
        self.lif1 = snn.Leaky(beta=b1)
        self.lif2 = snn.Leaky(beta=b2)
        self.out  = nn.Linear(M - 1, C)

    def forward(self, x):
        m1, m2 = self.lif1.init_leaky(), self.lif2.init_leaky()
        rec = []
        for _ in range(self.T):
            s1, m1 = self.lif1(self.fc(x),  m1)
            s2, m2 = self.lif2(s1, m2)
            rec.append(s2)
        S = torch.stack(rec, dim=-1)
        return self.out(rpc_embedding(S, self.T)), S

# ════════════════════════════════════════════
#  Train + Eval
# ════════════════════════════════════════════
def train(C, M, T, epochs=100, lr=1e-3, verbose=True):
    tr, va = make_loaders(C)
    model  = SNNOrderingFramework(C, M, C, T)
    opt    = torch.optim.Adam(model.parameters(), lr=lr)
    log    = defaultdict(list)

    for ep in range(epochs):
        model.train(); ep_loss = 0
        for x, y in tr:
            opt.zero_grad()
            lg, _ = model(x)
            loss  = F.cross_entropy(lg, y)
            loss.backward(); opt.step()
            ep_loss += loss.item()

        model.eval(); ok = tot = 0
        sc_per_cls = defaultdict(list)
        with torch.no_grad():
            for x, y in va:
                lg, S = model(x)
                ok  += (lg.argmax(1) == y).sum().item(); tot += len(y)
                for i, c in enumerate(y.tolist()):
                    sc_per_cls[c].append(S[i].sum().item())

        log['loss'].append(ep_loss / len(tr))
        log['acc'].append(ok / tot)
        log['sc'].append({c: sum(v)/len(v) for c, v in sc_per_cls.items()})

        if verbose and ep % 10 == 0:
            print(f"  ep={ep:3d}  loss={log['loss'][-1]:.4f}  acc={log['acc'][-1]:.3f}  sc={log['sc'][-1]}")

    return model, log, va

# ════════════════════════════════════════════
#  Entry point
# ════════════════════════════════════════════
if __name__ == '__main__':
    print("═══ E1: Ordering Separability (C=M=4) ═══")
    m, log, va = train(C=4, M=4, T=16)
    print(f"Final acc: {log['acc'][-1]:.3f}")

    print("\n═══ E3: Scaling ═══")
    for M in [4, 6, 8, 10]:
        T = max(16, M*2)
        print(f"\nM={M}  M!={math.factorial(M):,}  T={T}")
        _, log, _ = train(C=M, M=M, T=T, verbose=False)
        print(f"  → final acc = {log['acc'][-1]:.3f}")
執行順序建議

先確認 E1 的 loss 下降且 acc 高於 1/C（random chance）。確認後執行診斷工具確認 ordering 確實在起作用（非 firing rate）。E2 使用 E1 的 trained model 直接分析，不需重新訓練。E3 需要最多計算資源，M=20 時建議開 GPU。
